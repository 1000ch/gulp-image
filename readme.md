## Install

```sh
$ npm install --save-dev gulp-image
```

## Usage

This is an example of `gulpfile.js`.

```js
var gulp = require('gulp');
var image = require('gulp-image');

gulp.task('image', function () {
  gulp.src('./fixtures/*')
    .pipe(image())
    .pipe(gulp.dest('./dest'));
});

gulp.task('default', ['image']);
```
